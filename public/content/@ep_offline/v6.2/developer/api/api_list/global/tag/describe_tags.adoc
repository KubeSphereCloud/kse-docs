=== Method

== 请求示例
[,javascript]
----
{
  "action": "DescribeTags",
  "limit": 2,
  "offset": 0,
  "owner": [],
  "resources": [],
  "scope": "zone",
  "tag_key": null,
  "tag_name": "",
  "tags": [],
  "types": [],
  "verbose": 0
}
----
