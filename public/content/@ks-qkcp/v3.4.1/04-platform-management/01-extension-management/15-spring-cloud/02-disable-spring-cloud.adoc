. 在集群配置文件中，将 **spec:springcloud:enabled** 参数设置为 **false**，然后保存文件。
+
--
// YAML
include::../../../../../_ks_components/code/yaml.adoc[]

spec:
  springcloud:
    enabled: false

include::../../../../../_ks_components/code/codeEnd.adoc[]
--

. 执行以下命令删除 Spring Cloud 相关系统资源：
+
--
// Bash
include::../../../../../_ks_components/code/bash.adoc[]

helm uninstall -n kubesphere-spring-cloud-system springcloud-controller

include::../../../../../_ks_components/code/codeEnd.adoc[]

// Bash
include::../../../../../_ks_components/code/bash.adoc[]

kubectl delete crd configurationhistories.springcloud.kubesphere.io configurations.springcloud.kubesphere.io gatewayrouteconfigs.springcloud.kubesphere.io gateways.springcloud.kubesphere.io
