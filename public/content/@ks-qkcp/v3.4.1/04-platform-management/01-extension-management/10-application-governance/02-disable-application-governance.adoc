. 在集群配置文件中，将 **spec:servicemesh:enabled** 参数设置为 **false**，然后保存文件。
+
--
// YAML
include::../../../../../_ks_components/code/yaml.adoc[]

spec:
  servicemesh
    enabled: false

include::../../../../../_ks_components/code/codeEnd.adoc[]
--

. 在 **kubectl** 对话框，执行以下命令删除应用治理相关资源：
+
--
// Bash
include::../../../../../_ks_components/code/bash.adoc[]

curl -L https://istio.io/downloadIstio | sh -

include::../../../../../_ks_components/code/codeEnd.adoc[]

// Bash
include::../../../../../_ks_components/code/bash.adoc[]

istioctl x uninstall --purge

include::../../../../../_ks_components/code/codeEnd.adoc[]

// Bash
include::../../../../../_ks_components/code/bash.adoc[]

kubectl -n istio-system delete kiali kiali

include::../../../../../_ks_components/code/codeEnd.adoc[]

// Bash
include::../../../../../_ks_components/code/bash.adoc[]

helm -n istio-system delete kiali-operator

include::../../../../../_ks_components/code/codeEnd.adoc[]

// Bash
include::../../../../../_ks_components/code/bash.adoc[]

kubectl -n istio-system delete jaeger jaeger

include::../../../../../_ks_components/code/codeEnd.adoc[]

// Bash
include::../../../../../_ks_components/code/bash.adoc[]

helm -n istio-system delete jaeger-operator
