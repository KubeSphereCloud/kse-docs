. 安装 httpd。
+
[,bash]
----
yum install httpd-tools
----

. 生成认证文件。
+
--
[,bash]
----
mkdir -p /etc/kubekey/registry/auth
htpasswd -Bbn huawei Huawei@123 > /etc/kubekey/registry/auth/passwd
----

--

. 更新 systemd 配置。
+
--
[,bash]
----
# 查看 registry service 服务
systemctl status registry.service
----

image:/images/ks-qkcp/zh/security-handbook/config-username-password/3.png[]

[,bash]
----
# 修改 service 配置
vim /etc/systemd/system/registry.service

# 更新 Service 新增环境变量
[Service]
Environment="REGISTRY_AUTH=htpasswd"
Environment="REGISTRY_AUTH_HTPASSWD_REALM=Registry Realm"
Environment="REGISTRY_AUTH_HTPASSWD_PATH=/etc/kubekey/registry/auth/passwd"
----

image:/images/ks-qkcp/zh/security-handbook/config-username-password/4.png[]
--

. 重启 registry 服务。
+
--
[,bash]
----
systemctl daemon-reload && systemctl restart registry
----
