== 运维场景

通常情况下，系统盘为 sda 盘，在某些情况下，可能会识别错误，导致系统安装到其他磁盘上，可通过以下方式限制系统盘安装位置。

== 操作步骤

. 使用 u 盘安装系统。

. 修改 ``/preseed/custom.seed`` 文件，主要添加 ``BOOT_DISK="sda"; \`` 语句。
+
[source,shell]
----
vi /preseed/custom.seed
----
+
[source,shell]
----
# Partitioning settings
## Automatically select the first disk as the system disk
d-i partman/early_command string \
    HAS_SDA_DISK=$(lsblk -rndo NAME | egrep -q '^sda' && echo 'True' || echo 'False'); \
    SSD_DISK=$(lsblk -rndbo NAME,ROTA,SIZE,TRAN | egrep -v 'usb|sr0|sr1' | awk '{if($2 == 0 && $3 > 34359738368) print $1}' | sort -n | head -1); \
    SATA_DISK=$(lsblk -rndbo NAME,ROTA,SIZE,TRAN | egrep -v 'usb|sr0|sr1' | awk '{if($2 == 1 && $3 > 34359738368) print $1}' | sort -n | head -1); \
    if [[ "${HAS_SDA_DISK}" == "True" ]]; then \
        BOOT_DISK="$(echo `ls -l /sys/block/[hs]d* | grep -v "usb" | head -n1 | sed 's/^.*\([hs]d[a-z]\+\).*$/\1/'`)"; \
    else \
        if [[ -n "${SATA_DISK}" ]]; then \
            BOOT_DISK="${SATA_DISK}"; \
        elif [[ -n "${SSD_DISK}" ]]; then \
            BOOT_DISK="${SSD_DISK}"; \
        else \
            BOOT_DISK="sda"; \
        fi; \
    fi; \
    BOOT_DISK="sda"; \
     debconf-set partman-auto/disk "/dev/${BOOT_DISK}"; \
    debconf-set grub-installer/bootdev "/dev/${BOOT_DISK}"; \
    echo "HAS_SDA_DISK: ${HAS_SDA_DISK}"; \
    echo "BOOT_DISK: /dev/${BOOT_DISK}";
----
