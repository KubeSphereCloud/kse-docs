== 准备 ClickHouse 环境

本实践以 QingCloud ClickHouse 为例。

. link:../../../../../database/clickhouse/quickstart/create_cluster/[创建 ClickHouse 集群]。
. link:../../../../../database/clickhouse/quickstart/access_clickhouse/[连接 ClickHouse]。
. 连接成功后，执行以下命令，创建数据库 *pk*。
+
[,sql]
----
 create database if not exists pk;
----

. 执行以下命令，在数据库 *pk* 中创建数据库表 *output_uv*。
+
[,sql]
----
 create table output_uv
 (
     `userids`     String,
     `uv`          UInt64,
     `create_time` DateTime
 ) engine = SummingMergeTree(uv)
     primary key (userids)
     order by (userids);

 create table output_pv
 (
     `pagevisits` String,
     `product_id` String,
     `pv`         UInt64,
     `stt`        DateTime,
     `edt`        DateTime
 ) engine = Memory;

 CREATE TABLE `output_conversion_rate`
 (
     `conversion_rate` String,
     `rate`            String,
     `create_time`     DateTime
 ) engine = ReplacingMergeTree(create_time)
     partition by toYYYYMMDD(create_time)
     primary key (conversion_rate)
     order by (conversion_rate,create_time);
----
